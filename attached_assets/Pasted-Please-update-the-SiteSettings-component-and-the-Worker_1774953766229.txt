Social icon helper — use simple text symbols as fallback, or Unicode:
typescriptfunction socialIcon(platform: string): string {
  const icons: Record<string, string> = {
    instagram: '&#xe600;',  // use platform name as text fallback
    facebook: 'f',
    youtube: '▶',
    pinterest: 'P',
    tiktok: '♪',
  };
  // Preferred: use platform name styled as a pill button
  return `<span style="display:inline-flex;align-items:center;justify-content:center;width:36px;height:36px;border:1.5px solid #000;border-radius:50%;font-size:11px;font-weight:700;text-transform:uppercase;">${platform.charAt(0).toUpperCase()}</span>`;
}
